7. Construtores da super classe com argumentos


	public abstract class Musico{
		private String name;  //todos os músicos, inclusive as subclasses tem um nome.

		public String getName(){ //um método de captura que Cantor herda.
			return name;
		}
		public Musico(String theName){
			name = theName; //O construtor que usa o nome e o atribui à variável de instância name.
		}
	public class Cantor extends Musico{
		public Cantor(String Name){ //O construtor de Cantor usa um nome.
			super(name); // Envia o nome para cima na Pilha até o construtor de Musico.
		}
	}
	public class UseCantor{
		public static void main (String[] args){
			Cantor c = new Cantor("Bob Marley"); // Cria um objeto Cantor passando o nome Bob Marley para seu construtor.
			System.out.println(c.getName()); // Chama o metodo getName() herdado por Cantor. 
		}
	}
