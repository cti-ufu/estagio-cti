Não usar os valores de Enum para fazer contas matemáticas

	// Abuse of ordinal to derive an associated value - DON'T DO THIS
public enum Ensemble {
	SOLO,SEXTET,DUET,TRIO, QUARTET, QUINTET,SEPTET, OCTET, NONET, DECTET;
	public int numberOfMusicians() { 
		return ordinal() + 1; 
	}
}
Nunca determinar um valor associado com um enum da sua ordinal; armazená-lo em um campo de instância em vez disso: 

public enum Ensemble {
	SOLO(1), DUET(2), TRIO(3), QUARTET(4), QUINTET(5),SEXTET(6), 
	SEPTET(7), OCTET(8), DOUBLE_QUARTET(8),NONET(9), DECTET(10), 
	TRIPLE_QUARTET(12);
	private final int numberOfMusicians;

	Ensemble(int size) { 
		this.numberOfMusicians = size; 
	}
	public int numberOfMusicians() { 
		return numberOfMusicians; 
	}
}
