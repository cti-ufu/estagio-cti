Annotations usam o @ junto de seu nome para ser identificado como um Annotation.
Pode-se colocar os annotations sobre as classes, interfaces, métodos, parametros dos métodos, campos e variáveis locais.
@Entity
public class Vehicle {

    @Persistent
    protected String vehicleName = null;


    @Getter
    public String getVehicleName() {
        return this.vehicleName;
    }

    public void setVehicleName(@Optional vehicleName) {
        this.vehicleName = vehicleName;
    }
}
